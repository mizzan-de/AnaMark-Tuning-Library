178  }

180  // Read from stream, until '\n', '\r', '\0' or EOF is reached
181  while ( istr )
182  {
183  char ch = '\0';
184  istr.read(&ch, 1);
185  if ( (ch == '\0') || (ch == '\r') || (ch == '\n') )
186  {
187  // Remember first found EOL char to trigger line count correctly
188  if ( m_chEOL == '@' )
189  m_chEOL = ch;
190  // Increase line count if trigger character is found
191  if ( ch == m_chEOL )
192  lCurrLineCount = ++m_lLineCount;
193  break; // Done. Line end character reached
194  }
195  else
196  m_strLine.append(1, ch);
197  }
198 
199  strx::Trim(m_strLine);
200 
201  return true;
