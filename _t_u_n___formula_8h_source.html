167  // Set formula by given string
168  // returns false on error
169  bool SetFromStr(std::string strFormula)
170  {
171  Reset();
172 
173  strx::RemoveSpaces(strFormula);
174 
175  std::string::size_type pos = 0;
176  while ( pos < strFormula.size() )
177  {
178  switch ( strFormula.at(pos++) )
179  {
180  case '!': // Ensure_Hz
181  // Ensure_Hz overwrites everything else in this formula!
182  Reset();
183  if ( !strx::Eval(strFormula, pos, m_dblEnsureHz) )
184  return false;
185  pos = strFormula.size();
186  break;
187  case '#': // f_Range_Hz
188  if ( !m_rvpRangeHz.SetFromStr(strFormula, pos) )
189  return false;
190  break;
191  case '*': // MUL
192  if ( !strx::Eval(strFormula, pos, m_dblMUL) )
193  return false;
194  break;
195  case '/': // DIV
196  if ( !strx::Eval(strFormula, pos, m_dblDIV) )
197  return false;
198  break;
199  case '%': // CENTS
200  if ( !strx::Eval(strFormula, pos, m_dblCENTS) )
201  return false;
202  break;
203  case '+': // f_Shift_Hz
204  if ( !m_rvpShiftHz.SetFromStr(strFormula, pos) )
205  return false;
206  break;
207  case '~': // Loop function
208  if ( !strx::Eval(strFormula, pos, m_lLoop) )
209  return false;
210  if ( m_lMyIndex + m_lLoop < 0 )
211  m_lLoop = -GetOpenLoopValue();
212  if ( m_lMyIndex + m_lLoop >= MaxNumOfNotes )
213  m_lLoop = GetOpenLoopValue();
214  break;
215  default: // Unknown -> syntax error
216  return false;
217  }
218  }
219 
220  return true;
221  }

254  // Apply formula to vector of note frequencies
255  void Apply(std::vector<double> & vdblNoteFrequenciesHz) const
256  {
257  // Gets the sign of the amount of times to loop
258  // The sign will determine which direction to loop
259  long lInc = ( m_lLoop >= 0 ? +1 : -1 );
260  long l = 0;
261  do
262  {
263  // The current note index in regards to the entire scale
264  long scaleNoteIndex = m_lMyIndex + l;
265  // scaleNoteIndex can never be greater than MaxNumOfNotes, as vector
266  // vdblNoteFrequenciesHz's size is of MaxNumOfNotes
267  if ( (scaleNoteIndex < 0) || (scaleNoteIndex >= MaxNumOfNotes) )
268  break;
269 
270  // Shift entire scale instead of setting it, as the ! token has been used
271  if ( m_dblEnsureHz > 0 )
272  {
273  double dblFactor = m_dblEnsureHz / vdblNoteFrequenciesHz.at(scaleNoteIndex);
274  for ( long i = 0 ; i < MaxNumOfNotes ; ++i )
275  vdblNoteFrequenciesHz.at(i) *= dblFactor;
276  }
277  else // ! token hasn't been used.
278  {
279  // First resolve RV-Parameters (token #)
280  // If #= : Gets the frequency of another note in the scale,
281  // where the index was a single number (Absolute reference)
282  // If #> : Gets the frequency of another note in the scale,
283  // where that index is the current note index offset by an integer
284  // (Relative reference)
285  double dblRangeHz = m_rvpRangeHz.GetValue(vdblNoteFrequenciesHz, scaleNoteIndex);
286  // Token +
287  double dblShiftHz = m_rvpShiftHz.GetValue(vdblNoteFrequenciesHz, scaleNoteIndex);
288 
289  // Calculate
290  vdblNoteFrequenciesHz.at(scaleNoteIndex) =
291  dblRangeHz * m_dblMUL / m_dblDIV * pow(2, m_dblCENTS/1200) + dblShiftHz;
292  }
293  l += lInc;
294  } while ( std::abs(l) < std::abs(m_lLoop) );
295  }
